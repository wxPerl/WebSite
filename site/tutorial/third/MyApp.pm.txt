package MyApp;

use strict;
use vars qw(@ISA);

@ISA=qw(Wx::App);

use Wx qw(wxDefaultPosition);

sub OnInit {
    my( $this ) = shift;
    
    my( $frame ) = MyFrame->new( "Minimal wxPerl app",
                                 wxDefaultPosition,
                                 [450, 350]
                                 );
    
    $this->SetTopWindow( $frame );
    
    $frame->Show( 1 );
}

1;
